21  package com.github.klieber.phantomjs.archive;
22  
23  
24  public class PhantomJSArchiveBuilder {
25  	
26  	private final String platform;
27  	private final String arch;
28  	private final String version;
29  	
30  	public PhantomJSArchiveBuilder(String platform, String arch, String version) {
31  		this.platform = platform;
32  		this.arch     = arch;
33  		this.version  = version;
34  	}
35  	
36  	public PhantomJSArchiveBuilder(String version) {
37  		this(
38  				System.getProperty("os.name").toLowerCase(),
39  				System.getProperty("os.arch").toLowerCase(),
40  				version
41  		);
42  	}
43  	
44  	public PhantomJSArchive build() {
45  		PhantomJSArchive archive = null;
46    	if (platform.contains("win")) {
47    		archive = new WindowsPhantomJSArchive(version);
48    	} else if (platform.contains("mac")) {
49    		archive = new MacOSXPhantomJSArchive(version);
50    	} else if (platform.contains("nux")) {
51    		String modifier = arch.contains("64") ? "x86_64" : "i686"; 
52   			archive = new LinuxPhantomJSArchive(version, modifier);
53    	}
54    	if (archive == null) {
55    		throw new IllegalArgumentException("unknown platform: " + platform);
56    	}
57    	return archive;
58  	}
59  }
